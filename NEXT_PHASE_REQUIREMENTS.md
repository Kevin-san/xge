### 3.2 Mat4 逆矩阵算法改进

**当前问题：** `engine-math/src/mat4.rs#L175` 手写逆矩阵使用伴随矩阵 (adjugate)：

1. `inv[0][0] = m[1][1] * m[2][2] * m[3][3] - ... （4×4 伴随矩阵推导正确但**列/行混淆导致；行列式符号错误！**矩阵元素布局错误；导致符号错误（cols 是列主序但算法假设行主序。

**修复方案：**

```
```rust
impl Mat4 {
    /// 列主序伴随矩阵正确实现
    #[inline]
    pub fn inverse(&self) -> Option<Self> {
        let m = &self.cols; // cols[col][row]
        // 标准伴随矩阵：a00 a10 a20 a30
        //               a01 a11 a21 a31
        //               a02 a12 a22 a32
        //               a03 a13 a23 a33
        let a00 = m[0][0], a01 = m[0][1], a02 = m[0][2], a03 = m[0][3];
        let a10 = m[1][0], a11 = m[1][1], a12 = m[1][2], a13 = m[1][3];
        let a20 = m[2][0], a21 = m[2][1], a22 = m[2][2], a23 = m[2][3];
        let a30 = m[3][0], a31 = m[3][1], a32 = m[3][2], a33 = m[3][3];

        let b00 = a00 * a11 - a01 * a10;
        let b01 = a00 * a12 - a02 * a10;
        let b02 = a00 * a13 - a03 * a10;
        let b03 = a01 * a12 - a02 * a11;
        let b04 = a01 * a13 - a03 * a11;
        let b05 = a02 * a13 - a03 * a12;
        let b06 = a20 * a31 - a21 * a30;
        let b07 = a20 * a32 - a22 * a30;
        let b08 = a20 * a33 - a23 * a30;
        let b09 = a21 * a32 - a22 * a31;
        let b10 = a21 * a33 - a23 * a31;
        let b11 = a22 * a33 - a23 * a32;

        let det = b00 * b11 - b01 * b10 + b02 * b09 + b03 * b08 - b04 * b07 + b05 * b06;
        if det.abs() < 1e-6 { return None; }
        let inv_det = det.recip();

        Some(Self {
            cols: [
                [(a11 * b11 - a12 * b10 + a13 * b09) * inv_det,
                (a02 * b10 - a01 * b11 - a03 * b09) * inv_det,
                (a31 * b05 - a32 * b04 + a33 * b03) * inv_det,
                (a22 * b04 - a21 * b05 - a23 * b03) * inv_det,
                ...
            ]
        })
    }
}
```
